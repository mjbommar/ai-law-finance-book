% Scattered particles
\foreach \i in {1,...,100} {
  \pgfmathsetmacro{\px}{random()*8.5}
  \pgfmathsetmacro{\py}{random()*11}
  \pgfmathsetmacro{\psize}{0.4 + random()*1.0}
  \pgfmathsetmacro{\popacity}{0.08 + random()*0.25}
  \fill[GlowCyan, opacity=\popacity] (\px in, \py in) circle (\psize pt);
}

% Motion blur streaks
\begin{scope}[shift={(4.25in, 6.3in)}]
  \foreach \angle in {20, 55, 95, 130, 165, 200, 235, 275, 315, 350} {
    \pgfmathsetmacro{\len}{1.5 + random()*1.0}
    \pgfmathsetmacro{\startdist}{2.0}
    \pgfmathsetmacro{\sx}{\startdist*cos(\angle)}
    \pgfmathsetmacro{\sy}{\startdist*sin(\angle)*0.75}
    \pgfmathsetmacro{\ex}{(\startdist+\len)*cos(\angle)}
    \pgfmathsetmacro{\ey}{(\startdist+\len)*sin(\angle)*0.75}
    \draw[GlowTeal, opacity=0.2, line width=2.5pt, line cap=round]
      (\sx in, \sy in) -- (\ex in, \ey in);
    \draw[GlowCyan, opacity=0.1, line width=5pt, line cap=round]
      (\sx in, \sy in) -- (\ex in, \ey in);
  }
\end{scope}

% HIERARCHICAL FLOWCHART NETWORK
\begin{scope}[shift={(4.25in, 6.3in)}]

    \draw[WireColor, opacity=0.5, line width=1.2pt] (0, 0.1in) -- (\ex in, \ey in);
  }

  % Level 1 to Level 2 connections
  % From top-left node (120)
  \draw[WireColor, opacity=0.4, line width=1pt]
    ({1.1*cos(120)} in, {1.1*sin(120)*0.7} in) -- ({1.85*cos(150)} in, {1.85*sin(150)*0.7} in);
  \draw[WireColor, opacity=0.4, line width=1pt]
    ({1.1*cos(120)} in, {1.1*sin(120)*0.7} in) -- ({1.85*cos(100)} in, {1.85*sin(100)*0.7} in);

  % From right node (0)

    ({1.1*cos(0)} in, {1.1*sin(0)*0.7} in) -- ({1.85*cos(30)} in, {1.85*sin(30)*0.7} in);
  \draw[WireColor, opacity=0.4, line width=1pt]
    ({1.1*cos(0)} in, {1.1*sin(0)*0.7} in) -- ({1.85*cos(-30)} in, {1.85*sin(-30)*0.7} in);

  % From bottom node (240)
  \draw[WireColor, opacity=0.4, line width=1pt]
    ({1.1*cos(240)} in, {1.1*sin(240)*0.7} in) -- ({1.85*cos(210)} in, {1.85*sin(210)*0.7} in);
  \draw[WireColor, opacity=0.4, line width=1pt]
    ({1.1*cos(240)} in, {1.1*sin(240)*0.7} in) -- ({1.85*cos(270)} in, {1.85*sin(270)*0.7} in);

    ({1.85*cos(100)} in, {1.85*sin(100)*0.7} in) -- ({1.85*cos(30)} in, {1.85*sin(30)*0.7} in);
  \draw[WireColor, opacity=0.2, line width=0.6pt, dashed]
    ({1.85*cos(210)} in, {1.85*sin(210)*0.7} in) -- ({1.85*cos(150)} in, {1.85*sin(150)*0.7} in);

  \fill[NodeCore] (0, 0.1in) circle (5pt);

    \pgfmathsetmacro{\nx}{1.1*cos(\angle)}
    \pgfmathsetmacro{\ny}{1.1*sin(\angle)*0.7}
    % Glow layers

    \fill[GlowTeal, opacity=0.3] (\nx in, \ny in) circle (11pt);
    \fill[GlowCyan, opacity=0.75] (\nx in, \ny in) circle (7pt);
    \fill[NodeCore] (\nx in, \ny in) circle (3.5pt);
  }

  \foreach \angle in {150, 100, 30, -30, 210, 270} {
    \pgfmathsetmacro{\nx}{1.85*cos(\angle)}
    \pgfmathsetmacro{\ny}{1.85*sin(\angle)*0.7}
    % Diamond shape with glow
    \fill[GlowTeal, opacity=0.15] (\nx in, \ny in) circle (10pt);
    \fill[GlowCyan, opacity=0.35] (\nx in, \ny in) circle (6pt);
    \fill[NodeCore, opacity=0.9] (\nx in, \ny in) circle (3pt);
  }

  % ===== OUTER DATA RING =====
  \foreach \i in {0,...,23} {
    \pgfmathsetmacro{\angle}{\i * 15}
    \pgfmathsetmacro{\nx}{2.5*cos(\angle)}
    \pgfmathsetmacro{\ny}{2.5*sin(\angle)*0.7}
    \fill[GlowCyan, opacity=0.25] (\nx in, \ny in) circle (2pt);
  }

\end{scope}
