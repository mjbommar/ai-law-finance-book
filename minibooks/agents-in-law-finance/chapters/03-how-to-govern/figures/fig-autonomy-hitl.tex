\textcolor{slate-700}{$\blacktriangleright$} Actions are high-consequence and irreversible (filing court documents, executing large trades, signing contracts)\\[3pt]
\textcolor{slate-700}{$\blacktriangleright$} Professional duties require human judgment (attorney competence, fiduciary duty)\\[3pt]
\textcolor{slate-700}{$\blacktriangleright$} Regulatory requirements mandate human review (certain medical diagnoses, credit decisions)

    align=left
] at ([xshift=0.15cm, yshift=-1.4cm]card.north) {
\textbf{\textsf{CONTROL REQUIREMENTS}}\\[4pt]
\textbf{Primary Control:} Human reviewer approves every significant action\\[3pt]
\textbf{Logging:} Capture recommendation + human decision (approve/reject/modify)\\[3pt]
\textbf{Monitoring:} Lighter post-action review (human serves as primary control)\\[3pt]
\textbf{Key Risk:} Automation bias---humans rubber-stamping without meaningful review
};
